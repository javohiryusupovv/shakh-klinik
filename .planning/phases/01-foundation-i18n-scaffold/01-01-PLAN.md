**Step 3 — Create `app/[locale]/layout.tsx` (per D-07, D-10, D-11, I18N-06, PITFALLS §2):**

PITFALL §2: `params` is a Promise in Next 16. MUST `await params`.
D-10: `const { locale } = await params` — not `params.locale` directly.
D-11: `generateStaticParams` returns all 3 locales to lock them at build.
I18N-06: Set `<html lang={locale}>` — the locale layout owns the lang attribute, root layout does not.

```typescript
import { NextIntlClientProvider } from 'next-intl'
import { getMessages } from 'next-intl/server'
import { hasLocale } from 'next-intl'
import { notFound } from 'next/navigation'
import { routing } from '@/i18n/routing'
import LanguageSwitcher from '@/components/shared/LanguageSwitcher'

export function generateStaticParams() {
  return routing.locales.map((locale) => ({ locale }))
}

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode
  params: Promise<{ locale: string }>
}) {
  const { locale } = await params  // D-10: await params (PITFALLS §2)

  // Validate locale — redirects to 404 if someone manually visits /xx/...
  if (!hasLocale(routing.locales, locale)) {
    notFound()
  }

  const messages = await getMessages()

  return (
    <html lang={locale} suppressHydrationWarning>
      <body>
        <NextIntlClientProvider messages={messages}>
          {/* Phase 1 Header stub — minimal, replaced in Phase 2 */}
          <header style={{ padding: '1rem', borderBottom: '1px solid #eee' }}>
            <LanguageSwitcher />
          </header>
          <main>{children}</main>
        </NextIntlClientProvider>
      </body>

```

IMPORTANT: The `<html>` and `<body>` tags here create a nested HTML document only in the RSC tree — Next.js merges the locale layout's `<html lang>` attribute with the root layout's `<html>` at render. This is intentional and correct Next 16 pattern. See the internationalization guide.

Actually — reconsider the structure: in Next 16 App Router, ONLY the root layout should have `<html>` and `<body>`. Nested layouts should NOT repeat them. The locale layout should use a React Fragment or `<>` as the return wrapper. The root layout already has `<html>` and `<body>`. The locale layout just wraps children.

Correct locale layout structure (no html/body):
```typescript
export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode
  params: Promise<{ locale: string }>
}) {
  const { locale } = await params

  if (!hasLocale(routing.locales, locale)) {
    notFound()
  }

  const messages = await getMessages()

  return (
    <NextIntlClientProvider messages={messages}>
      <header style={{ padding: '1rem', borderBottom: '1px solid #eee' }}>
        <LanguageSwitcher />
      </header>
      <main>{children}</main>
    </NextIntlClientProvider>
  )
}
```

**Step 5 — Create `components/shared/LanguageSwitcher.tsx` (per D-21, I18N-03):**
Client component — uses next-intl navigation hooks to switch locale while preserving the current path.

```typescript
'use client'

import { useLocale, useTranslations } from 'next-intl'
import { useRouter, usePathname } from 'next-intl/navigation'
import { routing } from '@/i18n/routing'

const LOCALE_LABELS: Record<string, string> = {
  ru: '🇷🇺 RU',
  uz: '🇺🇿 UZ',
  en: '🇬🇧 EN',
}

export default function LanguageSwitcher() {
  const locale = useLocale()
  const router = useRouter()
  const pathname = usePathname()

  function handleLocaleChange(newLocale: string) {
    router.replace(pathname, { locale: newLocale })
  }

  return (
    <nav aria-label="Language switcher" style={{ display: 'flex', gap: '0.5rem' }}>
      {routing.locales.map((loc) => (
        <button
          key={loc}
          onClick={() => handleLocaleChange(loc)}
          aria-current={loc === locale ? 'true' : undefined}
          style={{
            fontWeight: loc === locale ? 'bold' : 'normal',
            cursor: loc === locale ? 'default' : 'pointer',
            background: 'none',
            border: 'none',
            padding: '0.25rem 0.5rem',
          }}
        >
          {LOCALE_LABELS[loc] ?? loc.toUpperCase()}
        </button>
      ))}
    </nav>
  )
}
```

NOTE: `useRouter` and `usePathname` from `next-intl/navigation` require that `createNavigation` be called first in next-intl v3. Check the installed version. If these exports don't exist at `next-intl/navigation`, the pattern is:
```typescript
import { createNavigation } from 'next-intl/navigation'
const { useRouter, usePathname, Link } = createNavigation(routing)
```
Read `node_modules/next-intl/navigation.js` or the package README after installing to determine the correct import pattern for the installed version. Use the actual API, not the assumed one.
