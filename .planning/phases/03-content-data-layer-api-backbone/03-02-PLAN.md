```ts
import 'server-only'

const TOKEN = process.env.TELEGRAM_BOT_TOKEN
const CHAT_ID = process.env.TELEGRAM_CHAT_ID

if (!TOKEN || !CHAT_ID) {
  // Fail loudly at import time in dev so missing env is caught before first request.
  throw new Error('TELEGRAM_BOT_TOKEN and TELEGRAM_CHAT_ID must be set in .env.local')
}

export async function sendTelegramMessage(html: string): Promise<{ ok: boolean }> {
  try {
    const res = await fetch(`https://api.telegram.org/bot${TOKEN}/sendMessage`, {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({
        chat_id: CHAT_ID,
        text: html,
        parse_mode: 'HTML',
        disable_web_page_preview: true,
      }),
      // 8s timeout via AbortSignal.timeout (Node 20+; available in Next 16 runtime)
      signal: AbortSignal.timeout(8000),
    })
    return { ok: res.ok }
  } catch {
    return { ok: false }
  }
}
```

```ts
import { z } from 'zod'

// D-12: shared client + server source of truth.
export const PHONE_REGEX = /^\+998 \d{2} \d{3}-\d{2}-\d{2}$/

const localeSchema = z.enum(['ru', 'uz', 'en'])

export const appointmentSchema = z.object({
  name: z.string().min(2, 'name.tooShort').max(80, 'name.tooLong'),
  phone: z.string().regex(PHONE_REGEX, 'phone.invalid'),
  serviceSlug: z.string().min(1, 'service.required'),
  doctorSlug: z.string().optional().or(z.literal('')),
  preferredDate: z
    .string()
    .regex(/^\d{4}-\d{2}-\d{2}$/, 'date.invalid')
    .refine((d) => d >= new Date().toISOString().slice(0, 10), 'date.past'),
  locale: localeSchema,
})
export type AppointmentInput = z.infer<typeof appointmentSchema>

export const contactSchema = z.object({
  name: z.string().min(2, 'name.tooShort').max(80, 'name.tooLong'),
  phone: z.string().regex(PHONE_REGEX, 'phone.invalid'),
  email: z.union([z.string().email('email.invalid'), z.literal('')]).optional(),
  serviceSlug: z.string().optional().or(z.literal('')),
  message: z.string().min(10, 'message.tooShort').max(1000, 'message.tooLong'),
  locale: localeSchema,
})
export type ContactInput = z.infer<typeof contactSchema>

export const reviewSchema = z.object({
  name: z.string().min(2, 'name.tooShort').max(80, 'name.tooLong'),
  rating: z.number().int().min(1).max(5),
  serviceSlug: z.string().min(1, 'service.required'),
  doctorSlug: z.string().min(1, 'doctor.required'),
  text: z.string().min(20, 'review.tooShort').max(1500, 'review.tooLong'),

})
```

**Step 4 — Create `lib/rateLimit.ts`** (per D-15, B3):

```ts
// D-15: in-memory sliding-window. Single-instance limitation accepted for v1.
// On Vercel serverless, cold starts may reset the map — documented tradeoff.
const WINDOW_MS = 60_000
const MAX_REQUESTS = 5

const buckets = new Map<string, number[]>()

export function checkRateLimit(ip: string): { allowed: boolean } {
  const now = Date.now()
  const cutoff = now - WINDOW_MS
  const list = (buckets.get(ip) ?? []).filter((t) => t > cutoff)
  list.push(now)
  buckets.set(ip, list)
  return { allowed: list.length <= MAX_REQUESTS }
}
```

**`app/api/appointment/route.ts`** (per D-09 emoji 📅, RU labels):

```ts
import { appointmentSchema } from '@/lib/validations'
import { sendTelegramMessage } from '@/lib/telegram'
import { checkRateLimit } from '@/lib/rateLimit'

export async function POST(request: Request) {
  try {
    const ip = (request.headers.get('x-forwarded-for')?.split(',')[0]?.trim()) || 'unknown'
    if (!checkRateLimit(ip).allowed) {
      return Response.json({ error: 'rate_limited' }, { status: 429 })
    }

    const body = await request.json()
    const parsed = appointmentSchema.safeParse(body)
    if (!parsed.success) {
      return Response.json(
        { error: 'validation_failed', fieldErrors: parsed.error.flatten().fieldErrors },
        { status: 422 },
      )
    }

    const d = parsed.data
    const esc = (s: string) => s.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    const html = [
      '📅 <b>Новая заявка на приём</b>',
      '',
      `<b>Имя:</b> ${esc(d.name)}`,
      `<b>Телефон:</b> ${esc(d.phone)}`,
      `<b>Услуга:</b> ${esc(d.serviceSlug)}`,
      d.doctorSlug ? `<b>Врач:</b> ${esc(d.doctorSlug)}` : '<b>Врач:</b> любой свободный',
      `<b>Желаемая дата:</b> ${esc(d.preferredDate)}`,
      `<b>Локаль:</b> ${d.locale}`,
      '',
      `<i>Получено: ${new Date().toISOString()}</i>`,
    ].join('\n')

    const result = await sendTelegramMessage(html)
    if (!result.ok) {
      return Response.json({ error: 'telegram_failed' }, { status: 502 })
    }
    return Response.json({ success: true }, { status: 200 })
  } catch {
    return Response.json({ error: 'telegram_failed' }, { status: 502 })
  }
}
```

```tsx
// Live mask: digits-only -> "+998 XX XXX-XX-XX"
function formatPhone(value: string): string {
  const d = value.replace(/\D/g, '').replace(/^998/, '').slice(0, 9)
  if (d.length === 0) return '+998 '
  const p1 = d.slice(0, 2)
  const p2 = d.slice(2, 5)
  const p3 = d.slice(5, 7)
  const p4 = d.slice(7, 9)
  let s = `+998 ${p1}`
  if (p2) s += ` ${p2}`
  if (p3) s += `-${p3}`
  if (p4) s += `-${p4}`
  return s
}
```
