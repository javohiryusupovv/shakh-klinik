- **slug**: kebab-case unique identifier (e.g. `cardiology-consultation`, `karimov-sherzod-alisherovich`, `clinic-opens-new-pediatric-wing`). MUST be URL-safe and stable — Phase 4 generates routes from these.

- **photoQuery**: same idea, doctor-specific (e.g. `male doctor portrait white coat 40s`).
- **coverImageQuery**: same idea, news-article-specific (e.g. `modern hospital reception`).

// Dev-time integrity check — fails fast if a slug reference is wrong
if (process.env.NODE_ENV !== 'production') {
  const serviceSlugs = new Set(SERVICES.map(s => s.slug));
  const doctorSlugs = new Set(DOCTORS.map(d => d.slug));
  const departmentIds = new Set(DEPARTMENTS.map(d => d.id));
  for (const s of SERVICES) {
    if (!departmentIds.has(s.departmentId)) throw new Error(`Service ${s.slug} → unknown departmentId ${s.departmentId}`);
  }
  for (const d of DOCTORS) {
    if (!departmentIds.has(d.departmentId)) throw new Error(`Doctor ${d.slug} → unknown departmentId ${d.departmentId}`);
  }
